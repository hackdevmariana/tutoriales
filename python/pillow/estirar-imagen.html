Estirar una imagen con PILLOW

Podemos usar la biblioteca Pillow en Python para redimensionar una imagen manteniendo la altura o la anchura constante.

from PIL import Image

imagen = Image.open('path/to/image.jpg')

nueva_anchura = 800 
nueva_altura = 600

factor = nueva_anchura / float(imagen.size[0])
altura_redimensionada = int((float(imagen.size[1]) * float(factor)))
imagen_redimensionada = imagen.resize((nueva_anchura, altura_redimensionada), Image.ANTIALIAS)

imagen_redimensionada.save('path/to/imagen_redimensionada.jpg')

factor = nueva_altura / float(imagen.size[1])
anchura_redimensionada = int((float(imagen.size[0]) * float(factor)))
imagen_redimensionada = imagen.resize((anchura_redimensionada, nueva_altura), Image.ANTIALIAS)

imagen_redimensionada.save('path/to/imagen_redimensionada_altura.jpg')
